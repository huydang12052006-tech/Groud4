public class EnvironmentArea {

    protected String areaId;
    protected String areaName;
    protected int pollutionLevel;

    // Constructor mặc định
    public EnvironmentArea() {
    }

    // Constructor đầy đủ tham số
    public EnvironmentArea(String areaId, String areaName, int pollutionLevel) {
        this.areaId = areaId;
        this.areaName = areaName;
        this.pollutionLevel = pollutionLevel;
    }

    public String getAreaId() {
        return areaId;
    }

    public String getAreaName() {
        return areaName;
    }

    public int getPollutionLevel() {
        return pollutionLevel;
    }

    public void setPollutionLevel(int pollutionLevel) {
        this.pollutionLevel = pollutionLevel;
    }

    // Đánh giá mức độ ô nhiễm
    public String evaluatePollution() {
        if (pollutionLevel < 30) {
            return "Low pollution";
        } else if (pollutionLevel < 70) {
            return "Medium pollution";
        } else {
            return "High pollution";
        }
    }

    public void display() {
        System.out.println("Area ID: " + areaId);
        System.out.println("Area Name: " + areaName);
        System.out.println("Pollution Level: " + pollutionLevel);
        System.out.println("Evaluation: " + evaluatePollution());
    }
}

update()
draw()

👉 nên phải để ở:

GameObject

Không chỉ Pipe mới move
Bird cũng move

👉 nên phải tách ra:

Movable (interface)
+ move()

👉 Hiện tại Pipe:

vừa có dữ liệu
vừa có logic chung
vừa có hành vi di chuyển

⛔ Vi phạm:

Single Responsibility Principle (1 class = 1 nhiệm vụ chính)

✅ 3. Thiết kế đúng (rất quan trọng)
🔹 Class cha
GameObject
- x, y, width, height
+ update()
+ draw()
🔹 Interface
Movable
+ move()
🔹 Pipe đúng chuẩn
Pipe extends GameObject implements Movable
- gapY

+ Pipe(...)
+ update()   (override nếu cần)
+ move()     (từ Movable)
🎯 4. Hiểu bản chất (cực quan trọng để không sai nữa)
🔥 Nguyên tắc vàng:
1. Kế thừa (extends)

👉 dùng khi:

IS-A

Ví dụ:

Pipe IS-A GameObject
2. Interface

👉 dùng khi:

CAN-DO

Ví dụ:

Pipe CAN move
🧠 5. So sánh để nhớ lâu
Sai của m	Đúng
Nhét move vào Pipe	Đưa vào Movable
Lặp update/draw	Để ở GameObject
Không rõ kế thừa	extends rõ ràng
Thiếu interface	thêm Movable
⚡ 6. Nói kiểu đi thi (ăn điểm)

Nếu thầy hỏi:

👉 “Tại sao không để move() trong Pipe?”

“Vì move là hành vi chung của nhiều đối tượng, em tách ra interface Movable để đảm bảo tái sử dụng và đúng nguyên tắc thiết kế OOP.”

import java.awt.Graphics;

public class Bird extends GameObject implements Movable {
    private int velocityY;
    private int gravity;

    public Bird(int x, int y, int width, int height, int gravity) {
        super(x, y, width, height);
        this.gravity = gravity;
        this.velocityY = 0;
    }

    public void fly() {
        velocityY = -10;
    }

    @Override
    public void move() {
        velocityY += gravity;
        y += velocityY;
    }

    @Override
    public void update() {
        move();
    }

    @Override
    public void draw(Graphics g) {
        System.out.println("Draw Bird at (" + x + ", " + y + ")");
    }
}

        super(x, y, width, height);
        this.gapY = gapY;
        this.velocityX = velocityX;
    }

    @Override
    public void move() {
        x -= velocityX;
    }

    @Override
    public void update() {
        move();
    }

    @Override
    public void draw(Graphics g) {
        System.out.println("Draw Pipe at (" + x + ", " + y + "), gapY = " + gapY);
    }
}

import java.util.ArrayList;

public class AreaManager {
    private ArrayList<EnvironmentArea> list = new ArrayList<>();

    public void addArea(EnvironmentArea area) {
        list.add(area);
    }

    public void displayAll() {
        for (EnvironmentArea a : list) {
            a.display();
            System.out.println("--------------");
        }
    }

    public int countPolluted() {
        int count = 0;
        for (EnvironmentArea a : list) {
            if (a.getPollutionLevel() > 60) count++;
        }
        return count;
    }

    public EnvironmentArea maxPollution() {
        if (list.isEmpty()) return null;

        EnvironmentArea max = list.get(0);
        for (EnvironmentArea a : list) {
            if (a.getPollutionLevel() > max.getPollutionLevel()) {
                max = a;
            }
        }
        return max;
    }
}
